    // Create particles for the hero section
    function createParticles() {
      const container = document.getElementById('particles-container');
      if (!container) return;
      
      const containerWidth = container.offsetWidth;
      const containerHeight = container.offsetHeight;
      
      // Create particles
      for (let i = 0; i < 50; i++) {
        const particle = document.createElement('div');
        particle.classList.add('particle');
        
        // Random size between 2px and 6px
        const size = Math.random() * 4 + 2;
        particle.style.width = `${size}px`;
        particle.style.height = `${size}px`;
        
        // Random position
        const x = Math.random() * containerWidth;
        const y = Math.random() * containerHeight;
        particle.style.left = `${x}px`;
        particle.style.top = `${y}px`;
        
        // Random opacity
        particle.style.opacity = Math.random() * 0.5 + 0.1;
        
        container.appendChild(particle);
        
        // Animate particle
        animateParticle(particle);
      }
    }
    
    function animateParticle(particle) {
      // Random movement distance
      const xMove = Math.random() * 100 - 50;
      const yMove = Math.random() * 100 - 50;
      
      // Random duration
      const duration = Math.random() * 20 + 10;
      
      // Current position
      const currentX = parseFloat(particle.style.left);
      const currentY = parseFloat(particle.style.top);
      
      // Target position
      const targetX = currentX + xMove;
      const targetY = currentY + yMove;
      
      // Start time
      const startTime = Date.now();
      
      function animate() {
        const elapsed = (Date.now() - startTime) / 1000;
        const progress = Math.min(elapsed / duration, 1);
        
        // Ease in-out function
        const easeProgress = progress < 0.5 ? 2 * progress * progress : 1 - Math.pow(-2 * progress + 2, 2) / 2;
        
        // Calculate new position
        const newX = currentX + (targetX - currentX) * easeProgress;
        const newY = currentY + (targetY - currentY) * easeProgress;
        
        // Update position
        particle.style.left = `${newX}px`;
        particle.style.top = `${newY}px`;
        
        if (progress < 1) {
          requestAnimationFrame(animate);
        } else {
          // Start a new animation when this one completes
          animateParticle(particle);
        }
      }
      
      requestAnimationFrame(animate);
    }
    
    // Initialize particles when the page loads
    window.addEventListener('load', createParticles);
